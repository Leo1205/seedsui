单页模式

单页模式包含:section容器间动画跳转、返回的功能。它会监听来判断android返回按键，并提供相关回调，示例见page.html

HTML

<section id="page_0">
  <a href="#page_1" data-toggle="page">第二页</a>
</section>
<section id="page_1" data-animation="slideLeft" class="page">
  <a href="#page_2" data-toggle="page">第三页</a>
</section>
<section id="page_2" data-animation="slideRight" class="page">
  第三页
</section>
元素 属性 值 描述
容器元素 data-animation slideLeft(默认值) 左右
slideRight 右左
slideUp 上下
slideDown 下上
fade 淡入淡出
zoom 缩放
none 无动画
class page 用单页模式打开
按钮元素 href #id 连接等于page的ID
target _blank(默认值) 打开新页面
_self 替换当前页面
data-toggle page 用单页模式打开

Javascript

var page = new Page({
  "onLoad":function(Page){
    console.log("正在加载：");
  },
  "onOpenEnd":function(Page){
    console.log("已经打开：");
  },
  "onCloseEnd":function(Page){
    console.log("已经关闭：");
  }
});

var page=new Page(params)

params设置
参数 类型 默认值 描述
onLoad Callback 无 页面加载中
onOpenEnd Callback 无 页面加载完成
onCloseEnd Callback 无 页面关闭完成

示例

var page = new Page({
  "onLoad":function(Page){},
  "onCloseEnd":function(Page){}
});
page方法
方法 参数 描述
open String pageId 根据id打开传入的page
String animation(可不填) 可以设置动画 none | slideLeft | slideRight | slideUp | slideDown | fade | zoom
back 无 返回上一页
close String pageId 关闭指定的page
String animation(可不填) 可以设置动画 none | slideLeft | slideRight | slideUp | slideDown | fade | zoom

示例

page.open(pageId);
page.back();

        {
          text : '菜单一',
              handler : function(s){
                console.log(s.target);
                t.setText("您点击了菜单一");

//数字计数器
var countvalues = new CountValues({
    "onInput":function(e){
        //console.log("输入中");
    },
    "onInputOut":function(e){
        //console.log("超过数字限制时");
        var input=e.target;
        input.nextElementSibling.innerHTML=Math.round(e.maxLength-input.value.length);
        input.nextElementSibling.classList.add("color-tip");
    },
    "onInputIn":function(e){
        //console.log("没有超过数字限制时");
        var input=e.target;
        input.nextElementSibling.innerHTML="";
        input.nextElementSibling.classList.remove("color-tip");
    }
});

var scrollbox=document.body;
var scrollboxHeight=scrollbox.clientHeight;
function scrollToBottom(){
    scrollbox.scrollTop+=10;
    if(scrollbox.scrollTop+scrollboxHeight<scrollbox.scrollHeight){
        requestAnimationFrame(scrollToBottom);
    }
}
